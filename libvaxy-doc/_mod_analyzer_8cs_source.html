63  public T[] GetResources<T>()
64  {
65  Type resourceType = typeof(T);
66 
67  if (!typeToName.ContainsKey(resourceType))
68  throw new LibvaxyException("Tried to get an unknown mod resource type");
69 
70  if (resourceType == typeof(ModGore)) // ModGore is an extreme edge case
71  return Reflection.GetStaticField<IDictionary<int, ModGore>>(typeof(ModGore), "modGores")
72  .Where(kvp => kvp.Value.GetType().Assembly == mod.Code)
73  .Select(kvp => kvp.Value)
74  .ToArray() as T[];
75 
76  string fieldName = typeToName[resourceType];
77  object value = Reflection.GetInstanceField<object>(mod, fieldName);
78 
79  if (value == null) // should never be null, this check is redundant
80  return null;
81 
82  if (resourceType == typeof(ModRecipe))
83  return (value as IList<ModRecipe>).ToArray() as T[];
84 
85  if (resourceType == typeof(EquipTexture))
86  return (value as IDictionary<Tuple<string, EquipType>, EquipTexture>).Values.ToArray() as T[];
87 
88  return (value as IDictionary<string, T>).Values.ToArray();
