336  // will cache the provided MemberInfo into its respective cache dictionary
337  internal static void CacheInfo(MemberInfo info)
338  {
339  string key = FullMemberName(info);
340 
341  if (info is FieldInfo)
342  fieldCache[key] = (FieldInfo)info;
343  else if (info is PropertyInfo)
344  propertyCache[key] = (PropertyInfo)info;
345  else if (info is MethodInfo)
346  methodCache[key] = (MethodInfo)info;
347  else if (info is ConstructorInfo)
348  constructorCache[key] = (ConstructorInfo)info;
349  }

351  internal static string FullMemberName(this MemberInfo info)
352  {
353  string name = $"{info.ReflectedType.FullName}.{info.Name}";
354 
355  if (info is MethodBase methodBase)
356  {
357  Type returnType = info is MethodInfo ? ((MethodInfo)info).ReturnType : typeof(void);
358  name = FullMethodName(info.ReflectedType, info.Name, methodBase.GetParameters().Select(p => p.ParameterType).ToArray());
359  }

364  // format: AssemblyName.TypeName.MethodName[ParameterType1, ParameterType2, ParameterType3, ...]
365  // e.g the full name of System.Random.Next(int, int) is System.Random.Next[Int32, Int32]
366  // the name of every constructor is .ctor so e.g System.Random..ctor[Int32] stands for new Random(int)
367  // this method is ugly
368  internal static string FullMethodName(Type reflectedType, string methodName, params Type[] parameters)
369  => $"{reflectedType.FullName}.{methodName}[{string.Join(", ", parameters.Select(t => t.Name).ToArray())}]";
